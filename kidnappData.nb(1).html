# returns the elbow plot to determine number of K
elbow <- function(kidnappVector){
  vector <- kidnappVector
  vector <- vector[! vector %in% remove]
  vector <- na.exclude(vector)
  wss <- numeric(12)
  for(k in 1:12)
    wss[k] = sum(kmeans(vector, k, nstart = 25)$withinss)
  wssResults <- data.frame(k = c(1:12), wss = wss)
  plot <- ggplot(data = wssResults, mapping = aes(x = k, y = wss))+
    geom_point()+
    geom_line()
  return(plot)
}

# Clustering function, returns 
Clustering <- function(vectorName,k,varName,newName){
  kidnappVector <- select(kidnappData,varName,eventid)
  # remove unknowns -99 if found
  if ("-99" %in% vectorName){
    kidnappVector <- kidnappVector[-grep("-99",vectorName),]
  }
  kidnappVector <- na.omit(kidnappVector) 
  Cluster <- kmeans(kidnappVector[1],k,nstart = 25)
  kidnappVector$cluster <- as.factor(Cluster$cluster)
  names(kidnappVector)[3] <- newName
  return(kidnappVector)
}

TsRegion <- function(TSdata){
  newTSformat <- TSdata %>% 
    group_by(Date=floor_date(TSdata$date, "month"))
  newTSformat <- count(newTSformat,Date)
  #create time series object
  gtdTS <- ts(newTSformat$n, frequency = 12)
  #Decompose the time series 
  gtdDecomposition <- decompose(gtdTS)
  # Plot the decomposition
  plotDecomposition <- plot(gtdDecomposition)
  #Build ARIMA Model 
  arimaModel <- auto.arima(newTSformat$n)
  # Predict 12 months into the future   
  arimaForecast <- forecast (arimaModel,h = 60) # h is the frequency and 60 is for months 
  # VIsualize the forecast
  plotForecast <- plot(arimaForecast)
  
  Objects <- list(plotDecomposition,arimaForecast,plotDecomposition)

}
